int isQueueFull(struct queue *Q){
	if(Q->front==-1 && Q->rear==-1)
		return(0);
	else{
		if(isQueueFull(Q))
			return(Q->capacity);
		return(((Q->capacity)-(Q->front)+(Q->rear)+1)%(Q->capacity));
	}
}
